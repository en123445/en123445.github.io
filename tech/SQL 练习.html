准备数据​

  1. 学生表 Student(s_id,s_name,s_birth,s_sex) –学生编号,学生姓名, 出生年月,学生性别

  2. 课程表 Course(c_id,c_name,t_id) – –课程编号, 课程名称, 教师编号

  3. 教师表 Teacher(t_id,t_name) –教师编号,教师姓名

  4. 成绩表 Score(s_id,c_id,s_score) –学生编号,课程编号,分数

CREATE DATABASE schoolEx

-- 学生表
CREATE TABLE `Student`(
`s_id` VARCHAR(20),
`s_name` VARCHAR(20) NOT NULL DEFAULT '',
`s_birth` VARCHAR(20) NOT NULL DEFAULT '',
`s_sex` VARCHAR(10) NOT NULL DEFAULT '',
PRIMARY KEY(`s_id`)
);

-- 课程表
CREATE TABLE `Course`(
`c_id` VARCHAR(20),
`c_name` VARCHAR(20) NOT NULL DEFAULT '',
`t_id` VARCHAR(20) NOT NULL,
PRIMARY KEY(`c_id`)
);

-- 教师表
CREATE TABLE `Teacher`(
`t_id` VARCHAR(20),
`t_name` VARCHAR(20) NOT NULL DEFAULT '',
PRIMARY KEY(`t_id`)
);

-- 成绩表
CREATE TABLE `Score`(
`s_id` VARCHAR(20),
`c_id` VARCHAR(20),
`s_score` INT(3),
PRIMARY KEY(`s_id`,`c_id`)
);

-- 1、查询"01"课程比"02"课程成绩高的学生的信息及课程分数

SELECT Student.*, Score.c_id, Score.s_score
FROM Student
LEFT JOIN Score
ON Student.s_id = Score.s_id
WHERE Student.s_id IN (
SELECT a.s_id 
    FROM (SELECT * FROM Score WHERE c_id = '01') AS a 
    LEFT JOIN(SELECT * FROM Score WHERE c_id = '02' ) AS b
    ON a.s_id = b.s_id
    WHERE a.s_score > b.s_score OR b.s_id IS NULL
);

-- 17、按平均成绩从高到低显示所有学生的所有课程的成绩以及平均成绩

SELECT 
Score.s_id AS sid,
(SELECT s_score FROM Score WHERE c_id = '01' AND s_id = sid) AS '语文',
(SELECT s_score FROM Score WHERE c_id = '02' AND s_id = sid) AS '数学',
(SELECT s_score FROM Score WHERE c_id = '03' AND s_id = sid) AS '英语',
AVG(s_score) AS '平均1',
ROUND(AVG(s_score),0) AS '平均2' -- 四舍五入把数值字段舍入为指定的小数位数
FROM Score
GROUP BY sid
ORDER BY AVG(s_score) DESC;

-- 18.查询各科成绩最高分、最低分和平均分： -- 以如下形式显示：课程ID，课程name，最高分，最低分，平均分，及格率，中等率，优良率，优秀率 --及格为>=60，中等为：70-80（左闭右开），优良为：80-90（左闭右开），优秀为：>=90

SELECT 
s.c_id AS '课程ID', 
Course.c_name AS '课程name',
(SELECT MAX(s_score) FROM Score WHERE c_id = s.c_id) AS '最高分',
(SELECT MIN(s_score) FROM Score WHERE c_id = s.c_id) AS '最低分',
(SELECT AVG(s_score) FROM Score WHERE c_id = s.c_id) AS '平均分',
(SELECT 
    COUNT(CASE WHEN s_score >= 60 THEN s_score ELSE NULL END) / COUNT(1)
    FROM Score
    WHERE c_id = s.c_id) AS '及格率',
(SELECT 
    COUNT(CASE WHEN s_score >= 70 AND s_score < 80 THEN s_score ELSE NULL END) / COUNT(1)
    FROM Score
    WHERE c_id = s.c_id) AS '中等率',
(SELECT 
    COUNT(CASE WHEN s_score >= 80 AND s_score < 90 THEN s_score ELSE NULL END) / COUNT(1)
    FROM Score
    WHERE c_id = s.c_id) AS '优良率',
(SELECT 
    COUNT(CASE WHEN s_score>= 90 THEN s_score ELSE NULL END) / COUNT(1)
    FROM Score
    WHERE c_id = s.c_id) AS '优秀率'
FROM Score AS s -- 这里使用别名后，其他位置不能识别"Score"
JOIN Course
ON s.c_id = Course.c_id
GROUP BY s.c_id;

-- 19、按各科成绩进行排序，并显示排名 -- 19.1 查询每门课成绩最好的前两名

/*
sql中使用@来定义一个变量，同时使用@rank：= 123 来给@rank变量赋值123
1.选出Score中每一课对应的c_id, s_score, 成绩排序 <- 排序
2.定义变量@rank，@tmid(中间值)
3.if(A,B,C)在SQL中表示，如果A条件成立，那么执行B，否则执行C
  对每个记录，设置@tmid表示上一次记录的课号
  如果当前的课号和@tmid一样，那么该记录的@rank就加1，
  如果@tmid和当前的课号不一样，那么就说明这记录时对应课号的首条记录，那么rank为1 <- 写编号rank
*/

-- 纵向展示
SELECT r.c_id, s_score, rank
FROM
(SELECT 
    tmp.c_id, 
    tmp.s_score, 
    IF(tmp.c_id = @tmid, @rank := @rank + 1, @rank := 1) AS rank, 
    @tmid := tmp.c_id
    FROM 
        (SELECT c_id, s_score 
        FROM Score 
        GROUP BY c_id, s_score 
        ORDER BY c_id ASC,s_score DESC) AS tmp, -- step1
        (SELECT @rank := 0, @tmid := NULL) AS tmp1 -- step2
) AS r
WHERE s_score IS NOT NULL AND rank <= 2; -- 限定前两名

-- 横向展示
SELECT 
    c_id,
    (SELECT s_score FROM Score WHERE s1.c_id = c_id ORDER BY s_score DESC LIMIT 1) AS '第一',
    (SELECT s_score FROM Score WHERE s1.c_id = c_id ORDER BY s_score DESC LIMIT 1, 1) AS '第二'
FROM Score AS s1
GROUP BY s1.c_id;

-- 20、查询学生的总成绩并进行排名

SELECT tmp.s_id, tmp.sumS, (@rank := @rank + 1) AS rank
FROM 
    (SELECT s_id, SUM(s_score) AS sumS
        FROM Score
        GROUP BY s_id
        ORDER BY sumS DESC
    ) AS tmp;
SET @rank := 0;

SELECT Student.*, r.c_id, s_score, rank
FROM
(SELECT 
    tmp.s_id, 
    tmp.c_id, 
    tmp.s_score, 
    IF(tmp.c_id = @tmid, @rank := @rank + 1, @rank := 1) AS rank, 
    @tmid := tmp.c_id

        (SELECT s_id, c_id, s_score 
        FROM Score

-- 23、统计各科成绩各分数段人数：课程编号,课程名称,[100-85),[85-70),[70-60),[0-60]及所占百分比

SELECT 
    s.c_id AS '课程ID', 
    c_name AS '课程name',
    (SELECT 
        COUNT(CASE WHEN s_score > 85 AND s_score <= 100 THEN s_score ELSE NULL END)
        FROM Score
        WHERE c_id = s.c_id) AS '[100-85)',
    (SELECT 
        COUNT(CASE WHEN s_score > 85 AND s_score <= 100 THEN s_score ELSE NULL END) / COUNT(1)
        FROM Score
        WHERE c_id = s.c_id) AS '[100-85)百分比',
    (SELECT 
        COUNT(CASE WHEN s_score > 70 AND s_score <= 85 THEN s_score ELSE NULL END)
        FROM Score
        WHERE c_id = s.c_id) AS '[85-70)',
    (SELECT 
        COUNT(CASE WHEN s_score > 70 AND s_score <= 85 THEN s_score ELSE NULL END) / COUNT(1)
        FROM Score
        WHERE c_id = s.c_id) AS '[85-70)百分比',
    (SELECT 
        COUNT(CASE WHEN s_score > 60 AND s_score <= 70 THEN s_score ELSE NULL END)
        FROM Score
        WHERE c_id = s.c_id) AS '[70-60)',
    (SELECT 
        COUNT(CASE WHEN s_score > 60 AND s_score <= 70 THEN s_score ELSE NULL END) / COUNT(1)
        FROM Score
        WHERE c_id = s.c_id) AS '[70-60)百分比',
    (SELECT 
        COUNT(CASE WHEN s_score BETWEEN 0 AND 60 THEN s_score ELSE NULL END)
        FROM Score
        WHERE c_id = s.c_id) AS '[0-60]',
    (SELECT 
        COUNT(CASE WHEN s_score BETWEEN 0 AND 60 THEN s_score ELSE NULL END) / COUNT(1)
        FROM Score
        WHERE c_id = s.c_id) AS '[0-60]百分比'     
FROM Score AS s
JOIN Course
ON s.c_id = Course.c_id
GROUP BY Course.c_id;

INNER JOIN Score
ON Student.s_id = Score.s_id
WHERE Score.c_id IN (
    SELECT Course.c_id
    FROM Course
    JOIN Teacher
    ON Course.t_id = Teacher.t_id
    WHERE Teacher.t_name = '张三'
    )
ORDER BY Score.s_score DESC
LIMIT 1;

SELECT Student.*, Score.s_score
FROM Course
    JOIN Teacher
    ON Course.t_id = Teacher.t_id
    JOIN Score
    ON Score.c_id = Course.c_id
    JOIN Student
    ON Student.s_id = Score.s_id
WHERE Teacher.t_name = '张三'
ORDER BY Score.s_score DESC
LIMIT 1;

-- 41、查询不同课程成绩相同的学生的学生编号、课程编号、学生成绩

SELECT s1.s_id, s1.c_id, s1.s_score
FROM Score AS s1
JOIN Score AS s2
ON s1.c_id != s2.c_id
AND s1.s_id = s2.s_id
AND s1.s_score = s2.s_score
GROUP BY s1.s_id, s1.c_id;

或者
SELECT DISTINCT s1.s_id, s1.c_id, s1.s_score
FROM Score AS s1
JOIN Score AS s2
ON s1.c_id != s2.c_id
AND s1.s_id = s2.s_id
AND s1.s_score = s2.s_score;

--46、查询各学生的年龄 -- 按照出生日期来算，当前月日 < 出生年月的月日则，年龄减一

SELECT s_id, 
    s_name,
    s_birth,
    (CASE WHEN MONTH(CURRENT_DATE) < MONTH(s_birth) 
    THEN (YEAR(CURRENT_DATE) - YEAR(s_birth) -1) 
    ELSE (YEAR(CURRENT_DATE) - YEAR(s_birth)) 
    END) AS age
FROM Student;

或者
SELECT s_id, 
    s_name,
    s_birth,
    (DATE_FORMAT(NOW(), '%Y') - DATE_FORMAT(s_birth, '%Y') -
        (CASE 
        WHEN DATE_FORMAT(NOW(), '%m%d') > DATE_FORMAT(s_birth, '%m%d') THEN 0 
        ELSE 1 
        END)
    ) AS age
FROM Student;
